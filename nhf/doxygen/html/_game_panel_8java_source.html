241 public void run() {
242 double drawInterval = 1000000000/fps; // Ez egy másodperc nano sec-ben nézve és elosztva az fps-el, ennyi időnként fissítsuk a képernyöt
243 double nextDrawTime = System.nanoTime() + drawInterval; // nano sec-ben számolva mikor kell kövinenk frissíteni a képet
244 long timer = 0;
245 int drawCount = 0;
246
247 // A game loop addig megy ameddig tart a main szállunk
248 while(map_state != Map_Status.ERROR && gamer.getHealt() > 0) {
249
250 try {
251 updatePanel();
252 } catch (IOException e) {
253 e.printStackTrace();
254 }
255
256 repaint();
257 drawCount++;
258
259 try {
260 double remainingTime = nextDrawTime - System.nanoTime(); // maga a parancsok végrehajtása közben is eltelt idő, ha maradt azzal dolgozunk
261 remainingTime /= 1000000; // nano sec-et át kell váltani milli sec-re, mert a sleep-nek az kell
262 if(remainingTime < 0) {
263 remainingTime = 0; // Ha gond lenne, lagolnánk, akkor ne essen minuszba a sleep time
264 }
265 Thread.sleep((long)remainingTime);
266 nextDrawTime += drawInterval;
267
268 // Minden másodpercben kiírjuk, hányszor volt meghívva a repaint
269 if (System.nanoTime() - timer >= 1000000000) {
270 System.out.println("FPS: " + drawCount);
271 drawCount = 0;
272 timer = System.nanoTime();
273 }
274
275 } catch (InterruptedException e) {
276 this.errorIsHappened("In GamePanel run: Thread.sleep throw InterruptedException");
277 }
278 }
